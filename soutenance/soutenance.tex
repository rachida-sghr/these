\section{État de l'art}

%%%%%%%%%%%%%%%% Architecture SI
\begin{frame}
	\frametitle{Architecture d'Entreprise}

	\begin{block}{Définition}
		\justifying
		\itshape Une architecture d'entreprise est un ensemble pertinent 
d'artefacts de conception ou de représentations descriptives pour décrire une 
entreprise de manière à ce que cette entreprise soit créée en respectant certaines 
exigences et à ce qu'elle soit facilement maintenue tout au long de son cycle de 
vie.  \upshape \\(Zachman, 1997)  
	\end{block}
	
	\begin{block}{Cadres d'architecture -- Approches par points de vue}
		\begin{itemize}
			\item Séparer les préoccupations des différents acteurs
			\item Points de vue hiérarchisés : \itshape << IT follows Business >>
			\item \upshape Quatre points de vue identifiés : métier, fonctionnel,applicatif, technique
		\end{itemize}
	\end{block}
	
	\begin{center}
		\includegraphics[scale=0.23]{images/frameworks.png}
	\end{center}		
		
\end{frame}

\begin{frame}
	\frametitle{Limites des pratiques courantes d'Architecture d'Entreprise}
	\vspace{5mm} 
Des représentations informelles, hétérogènes et incompatibles avec la constante évolution de l'entreprise

%%%%%%%%%%%%%%%% Slide Modèle, métamodèle et Relation de l'IDM
\begin{frame}
	\frametitle{Ingénierie Dirigée par les Modèles}

		\begin{block}{Modèle -- Définition}
		\justifying
\itshape Un modèle est une \textbf{abstraction} d'un système, construite \textbf{selon le bon point de vue}, qui permet de répondre à des questions prédéfinies sur ce système en lieu et place de celui-ci. 
\upshape (Bézivin, Gerbé, 2001)
		\end{block}
		
		\begin{block}{Relations fondamentales de l'IDM (Favre 2004)}
			\begin{itemize}					
				\item $\chi$ :  \itshape ConformeÀ 
				\item $\mu$ : \itshape ReprésentationDe 
			\end{itemize}
		\end{block}
		
		\begin{center}
			\includegraphics[scale=0.25]{images/idm.png}
		\end{center}
		
		
\end{frame}


%%%%%%%%%%%%%%%% Slide Transformation de modèles et Model Typinge
\begin{frame}
	\frametitle{Ingénierie Dirigée par les Modèles}
	
	\begin{block}{Transformation de modèle (Mens, Van Gorp, 2006)}
		\justifying
			\begin{itemize}					
				\item Des modèles << contemplatifs >> aux modèles << productifs >>
				\item Construire des SI évolutifs, s'alignant rapidement sur le métier 
			\end{itemize}
	\end{block}
	
	\begin{center}
			\includegraphics[scale=0.4]{images/transformation.png}
	\end{center}
		
\end{frame}



\begin{frame}
	\frametitle{Langages exécutables pour l'Architecture d'Entreprise}

	\begin{block}{Avantages des langages exécutables :}
		
		\begin{itemize}
			\item Augmenter l'évolutivité des modèles en aidant à leur validation
			\item Lever les ambiguïtés des modèles purement contemplatifs

\section{Conclusion et Perspectives}
\begin{frame}
	\begin{block}{Conclusion}
		\begin{itemize}
			\item Contribution : Modélisation par points de vue du SI dans un cadre de \textbf{cohérence} par l'ajout d'un point de vue \textbf{Intégration} et le recours systématique aux langages \textbf{exécutables} pour la \textbf{simulation}
			\item Validation : application Smart Grids à travers la gestion d'une flotte de véhicules électriques
		\end{itemize} 
	\end{block}
	
		\begin{block}{Perspectives}
		\begin{itemize}
			\item Mise en \oe{}uvre du Model Typing pour l'implémentation de la vue Intégration
			\item Co-simulation des trois points de vue en pilotant la simulation du SI par le processus métier (implémentation avec Papyrus).
		\end{itemize} 
	\end{block}
\end{frame}
